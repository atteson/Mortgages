\begin{lem}
  \begin{eqnarray*}
    \P{X_T} = X_t + \int_0^T\P{\mub{X}_s}ds
  \end{eqnarray*}
\end{lem}
\begin{proof}
  \begin{eqnarray*}
    \P{X_T} & = & \P{X_t + \int_t^T \mub{X}_s ds + \int_t^T\sigmab{X}_s dW_s}\\
    & = & \P{X_t} + \P{\int_t^T \mub{X}_s ds} + \P{\int_t^T\sigmab{X}_s dW_s}\\
    & = & X_t + \P{\int_t^T \mub{X}_s ds} = X_t + \int_t^T \P{\mub{X}_s} ds\\
  \end{eqnarray*}
\end{proof}

\begin{thm}
  \begin{eqnarray*}
    \P{X_T}
  \end{eqnarray*}
\end{thm}
